package logic;

import java.util.InputMismatchException;
import java.util.Scanner;

public class run {
//Ejercicio corregido
	public static void main(String[] args) {

		boolean repeat = true;

		do {


			try {
				Scanner sc = new Scanner(System.in);
				System.out.println("***********");
				System.out.println("Digite el dividendo: ");
				int dividendo = sc.nextInt();
				System.out.println("Digite el divisor: ");
				int divisor = sc.nextInt();



				double resp = dividendo / divisor;
				System.out.println(resp);
				repeat = false;


			} catch (ArithmeticException e) {
				System.err.println("¡No se puede dividir por cero! \n");

			} catch (InputMismatchException e) {
				System.err.println("¡No se puede ingresar letras! \n");

			}
		} while (repeat);


	}

}
